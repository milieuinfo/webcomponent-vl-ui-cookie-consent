Cookie-consent met extra opt-in dat verplicht is

                    data-vl-opt-in-socialmedia-label="Sociale media"
                    data-vl-opt-in-socialmedia-description="Beschrijving van de sociale media cookies."
                    data-vl-opt-in-socialmedia-required></vl-cookie-consent>

Cookie-consent dynamisch extra opt-in toevoegen

            
                <vl-cookie-consent id="cookie-consent-6" data-vl-auto-open-disabled></vl-cookie-consent>
                <button is="vl-button" onclick="document.querySelector('#cookie-consent-6').addOptIn({
                    name: 'socialmedia',
                    label: 'Sociale media',
                    description: 'Sociale media beschrijving',
                    value: true,
                    required: false,
                    callbacks: {
                        activated: () => {},
                        deactivated: () => {}
                    }
                });">Voeg sociale media opt-in toe</button>
