                </vl-cookie-consent>

Cookie-consent dynamisch extra opt-in toevoegen

            
                <vl-cookie-consent id="cookie-consent-with-extra-dynamic-opt-in" data-vl-auto-open-disabled></vl-cookie-consent>
                <button id="button-open-cookie-consent-with-extra-dynamic-opt-in" is="vl-button" onclick="document.querySelector('#cookie-consent-with-extra-dynamic-opt-in').open();">Open cookie-consent</button>
                <button is="vl-button" onclick="document.querySelector('#cookie-consent-with-extra-dynamic-opt-in').addOptIn({
                    name: 'socialmedia',
                    label: 'Sociale media',
                    description: 'Sociale media beschrijving',
                    value: true,
                    mandatory: false,
                    callbacks: {
                        activated: () => {},
                        deactivated: () => {}
                    }
                });">Voeg sociale media opt-in toe</button>
